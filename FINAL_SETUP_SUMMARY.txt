Example 3: Complete RAG Pipeline
─────────────────────────────────
from utils.bedrock_helper import quick_setup
from langchain.chains import RetrievalQA
from langchain.schema import Document

# Setup
llm, embeddings, vectorstore = quick_setup("qa_system")

# Add documents
docs = [Document(page_content="Your text here")]
vectorstore.add_documents(docs)

# Create Q&A chain
qa = RetrievalQA.from_chain_type(
    llm=llm,
    retriever=vectorstore.as_retriever()
)

# Ask questions
answer = qa({"query": "Your question here"})
print(answer["result"])
