Example 2: Simple RAG Pipeline
───────────────────────────────
from utils.bedrock_helper import quick_setup
from langchain.chains import RetrievalQA
from langchain.schema import Document

# Setup
llm, embeddings, vectorstore = quick_setup("qa_system")

# Add documents
docs = [Document(page_content="Claude 3 is great for RAG!")]
vectorstore.add_documents(docs)

# Ask questions
qa = RetrievalQA.from_chain_type(llm=llm, retriever=vectorstore.as_retriever())
answer = qa({"query": "What's great for RAG?"})
print(answer["result"])
